function Register()
	{
		alert("Username: " + document.registerInfo.username.value + "\nPassword:" + document.registerInfo.password.value + "\nRE - Password:" + document.registerInfo.rePassword.value + "\nemail: " + document.registerInfo.email.value);
		var password = document.registerInfo.password.value;
		var rePassword = document.registerInfo.rePassword.value;
		
		if(password.match( rePassword))
		{
			var username = document.registerInfo.username.value;
			var password = document.registerInfo.password.value;
			var email = document.registerInfo.email.value;
			var user = new Parse.User();
			user.set("username", username);
			user.set("password", password);
			user.set("email", email);
			
			user.signUp(null, {
				  success: function(user) {
				  	alert("Success Finally!");
				  	window.location.href = 'MainMenu.html';
				    // Hooray! Let them use the app now.
				  },
				  error: function(user, error) {
				    // Show the error message somewhere and let the user try again.
				    alert("Error: " + error.code + "\n\nwhat is the error \n\n " + error.message);
				  }
			});
			
			
		}
		else
		{
			alert("Passwords do not match");
			
		}
	};
